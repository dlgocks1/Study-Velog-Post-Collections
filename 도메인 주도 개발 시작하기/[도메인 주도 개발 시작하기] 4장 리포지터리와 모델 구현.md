class Order(
    val id: OrderId,
    val orderer: Orderer,
    val orderLines: List<OrderLine>,
    var shippingInfo: ShippingInfo,
    var status: OrderStatus = OrderStatus.PAYMENT_WAITING,

            "이미 배송이 시작된 주문은 배송지를 변경할 수 없습니다"
        }
        shippingInfo = newShippingInfo

    fun cancel() {
        check(status == OrderStatus.PAYMENT_WAITING || status == OrderStatus.PREPARING) {

        status = OrderStatus.CANCELLED
    }

    fun completePayment() {
        check(status == OrderStatus.PAYMENT_WAITING) { "결제 대기 상태가 아닙니다" }
        status = OrderStatus.PREPARING
    }
}

```kotlin
// 도메인 리포지터리 구현
@Repository
class OrderRepositoryImpl(
    private val jpaRepository: OrderJpaRepository
) : OrderRepository {

    override fun save(order: Order): Order {
        // 신규 vs 수정 구분 - findById로 존재 여부 확인 후 update
        val entity = jpaRepository.findById(order.id.value)
            .map { existing -> existing.apply { updateFrom(order) } }
            ?: OrderJpaEntity.from(order)

        return jpaRepository.save(entity).toDomain()
    }

    override fun findById(id: OrderId): Order? =
        jpaRepository.findByIdOrNull(id.value)?.toDomain()

    override fun findByOrderer(memberId: Long): List<Order> =
        jpaRepository.findByOrdererMemberId(memberId).map { it.toDomain() }

    override fun findByStatus(status: OrderStatus): List<Order> =
        jpaRepository.findByStatus(status).map { it.toDomain() }

    override fun nextId(): OrderId =
        OrderId(System.currentTimeMillis()) // 실제론 Snowflake ID 등 사용
}

// update 편의 함수 (JpaEntity 내 혹은 확장함수로)
private fun OrderJpaEntity.updateFrom(order: Order) {
    shippingReceiverName  = order.shippingInfo.receiverName
    shippingReceiverPhone = order.shippingInfo.receiverPhone
    shippingZipCode       = order.shippingInfo.shippingAddress.zipCode
    shippingAddress1      = order.shippingInfo.shippingAddress.address1
    shippingAddress2      = order.shippingInfo.shippingAddress.address2
    shippingMessage       = order.shippingInfo.message
    status                = order.status
}
```

### 파사드, 헥사고날 구조와의 차이점

**DDD 관점: 도메인 모델 보호**

![](https://velog.velcdn.com/images/cksgodl/post/1cec78a2-99b2-4a49-8bfa-1bf6f99d5bd4/image.png)

-  Domain Layer:  UserRepository (interface) ← 도메인이 정의
- Infra Layer:   JpaUserRepository (impl)   ← 인프라가 구현
  - 도메인 계층이 인터페이스를 소유
  - 영속성 기술(JPA, Mongo, Redis)을 도메인에서 분리
  - "갈아끼울 수 있다" = 저장소 구현체를 바꿀 수 있다는 의미


**Facade 패턴 관점: 복잡한 서브시스템을 단순한 인터페이스로 감싸기**

![](https://velog.velcdn.com/images/cksgodl/post/bb6b4daf-f8af-4180-8693-4e3842bdebda/image.png)

```
Client → OrderFacade → { OrderService, PaymentService, InventoryService }
```
  - 여러 서비스/모듈의 복잡성을 숨기는 단일 진입점
  - 내부 구현을 갈아끼우는 게 목적이 아니라 사용을 단순화하는 게 목적
  - DDD의 Application Service가 사실상 Facade 역할을 함


**Hexagonal Architecture (Ports & Adapters) 관점: 애플리케이션 전체의 외부 의존성 격리 **

![](https://velog.velcdn.com/images/cksgodl/post/67d76f54-3034-4628-948b-72919e6ddf4b/image.png)

  - Port = 인터페이스 (in-port: 유스케이스, out-port: 외부 시스템) 
  - Adapter = 구현체 (HTTP, gRPC, JPA, Kafka 등)
   - 왼쪽이 driving adapter — 외부가 도메인을 호출. 오른쪽이 driven adapter — 도메인이 외부를 호출(하지만 인터페이스만 알고 있음).
   
> DDD Repository ↔ Hexagonal: 사실상 같은 것. 
> OrderRepository = driven port, OrderRepositoryImpl = driven adapter. 우리가 이미 구현한 게 헥사고날의 개념

####  관계 정리

```
  Hexagonal Architecture (가장 넓은 구조)
    └── DDD Repository = out-port 중 하나 (영속성 담당)
    └── in-port = Application Service ≈ Facade 역할

  Facade = 디자인 패턴 (구조적 단순화)
  DDD Repository = 도메인 패턴 (영속성 추상화)
  Hexagonal = 아키텍처 패턴 (전체 시스템 구조)
```

![](https://velog.velcdn.com/images/cksgodl/post/75172ffe-081b-4e54-8a73-e3b659213b88/image.png)

요약: DDD Repository는 "DB를 갈아끼울 수 있다", 헥사고날은 "DB든 메시지큐든 HTTP든 전부 갈아끼울 수 있다"로 범위가 확장된 것 Facade는 교체보다는 단순화가 목적이라 결이 다르다.

### EntityManager의 트랜잭션 내부에서의 update

 JPA에서는 명시적 update 메서드가 없다. 대신 `Dirty Checking(변경 감지)`로 동작
 
   핵심 흐름
```
  1. 트랜잭션 시작
  2. em.find() → DB에서 엔티티 조회 → 영속성 컨텍스트에 저장
  3. entity.setName("new") → 자바 객체만 수정 (SQL 없음)
  4. 트랜잭션 커밋 시점
     └→ flush() 자동 호출
        └→ 영속성 컨텍스트의 "스냅샷"과 현재 상태 비교
           └→ 변경된 필드 발견 → UPDATE SQL 생성 및 실행
```        

#### 영속성 컨텍스트란?

 DB와 애플리케이션 사이에 있는 1차 캐시 + 변경 추적기
 
```
  Application ←→ [ 영속성 컨텍스트 ] ←→ DB
                    (메모리 영역)
                    - 1차 캐시
                    - 스냅샷 저장
                    - 쓰기 지연 SQL 저장소
                    - 변경 감지
```

하는 일 4가지

1. 1차 캐시

```java
  User user1 = em.find(User.class, 1L);  // DB 조회 → 캐시에 저장
  User user2 = em.find(User.class, 1L);  // DB 안 감 → 캐시에서 반환

  user1 == user2  // true (같은 객체)
```

같은 트랜잭션 안에서 같은 엔티티를 두 번 조회하면 DB를 한 번만 조회

2. 동일성 보장

```java
  // 같은 @Id면 항상 같은 객체 참조를 반환
  em.find(User.class, 1L) == em.find(User.class, 1L)  // true
```

3. 쓰기 지연 (Write-behind)

```java
  em.persist(userA);  // INSERT SQL 안 나감, 저장소에 쌓아둠
  em.persist(userB);  // 마찬가지

  // 커밋 시점에 모아서 한번에 전송
  // → INSERT userA
  // → INSERT userB
```

4. 변경 감지 (Dirty Checking)

생명주기
```
  EntityManager 생성 = 영속성 컨텍스트 생성
         ↓
    트랜잭션 동안 엔티티 관리
         ↓
  EntityManager 종료 = 영속성 컨텍스트 소멸
                       (모든 엔티티 → 준영속 상태)
  Spring에서는 보통 트랜잭션 단위로 생성/소멸

  @Transactional 시작 → 영속성 컨텍스트 생성
  @Transactional 끝   → flush → commit → 영속성 컨텍스트 소멸
```

엔티티의 4가지 상태
```
   new User()          em.find()
       │                   │
       ▼                   ▼
    비영속(New)         영속(Managed) ←── 영속성 컨텍스트가 관리 중
       │                   │
       │ em.persist()      │ em.detach() / 트랜잭션 종료
       └──→ 영속 ──────────→ 준영속(Detached)
                           │
                           │ em.remove()
                           ▼
                        삭제(Removed)
```


![](https://velog.velcdn.com/images/cksgodl/post/565fc2dd-d1fc-4550-a4b1-66ec45654129/image.png)


JPA의 update는 "변경하라"고 명령하는 게 아니라, 영속 상태 객체를 수정하면 트랜잭션 커밋 시점에 자동으로 감지되어 SQL이 나가는 구조 (트랜잭션 내부에서만 구현)

## 4.3 매핑 구현

- **애그리거트 루트는 엔티티이므로 @Entity로 매핑한다.**
- 밸류는 @Embeddable로 매핑설정한다.
   - "나는 다른 엔티티에 내장될 수 있는 값 타입이야"
- 배률 타입 프로퍼티는 @Embedded로 매핑 설정한다.
   - "이 필드는 내장된 값 타입이야"
   

Value Object는 독자적 테이블이 필요 없다 → 소속된 엔티티의 테이블 컬럼으로 들어간다


```kotlin
// @Embeddable: 값 타입 클래스에 붙임
  @Embeddable
  class Address(
      val street: String,
      val city: String,
      val zipCode: String,
  )

  @Entity
  class Order(
      @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
      val id: Long = 0,

      val orderName: String,

      // @Embedded: 엔티티의 필드에 붙임
      @Embedded
      val shippingAddress: Address,
  )
```

테이블은 하나

```sql
  -- Order 테이블 하나에 Address 컬럼이 함께 들어감
  CREATE TABLE orders (
      id          BIGINT PRIMARY KEY,
      order_name  VARCHAR,
      street      VARCHAR,   ← Address의 필드
      city        VARCHAR,   ← Address의 필드
      zip_code    VARCHAR,   ← Address의 필드
  );
```

#### AttributeCOnverter활용한 밸류 매핑 처리

```kotlin
@Converter
class MoneyConverter : AttributeConverter<Money, String> {
    private val mapper = ObjectMapper()

    override fun convertToDatabaseColumn(attribute: Money?): String? =
        attribute?.let { mapper.writeValueAsString(it) }
        // {"amount":10000,"currency":"KRW"}

    override fun convertToEntityAttribute(dbData: String?): Money? =
        dbData?.let { mapper.readValue(it, Money::class.java) }
}
```

### 별도 테이블 매핑

#### `@ElementCollection` / `@CollectionTable`

Entity가 아닌 단순 값(또는 Embeddable)의 컬렉션을 별도 테이블에 저장할 때 쓰는 어노테이션이다.

```kotlin
@Entity
class Member(
    @Id @GeneratedValue
    val id: Long = 0,
    val name: String,

    // 단순 String 컬렉션
    @ElementCollection
    @CollectionTable(
        name = "member_nicknames",          // 생성될 테이블명
        joinColumns = [JoinColumn(name = "member_id")]  // FK 컬럼명
    )
    @Column(name = "nickname")
    val nicknames: MutableSet<String> = mutableSetOf(),

    // Embeddable 컬렉션
    @ElementCollection
    @CollectionTable(
        name = "member_addresses",
        joinColumns = [JoinColumn(name = "member_id")]
    )
    val addresses: MutableList<Address> = mutableListOf()
)

@Embeddable
data class Address(
    val city: String,
    val street: String,
    val zipcode: String
)

-- member_nicknames
member_id | nickname
----------+---------
1         | "해찬짱"
1         | "hchan"

-- member_addresses  
member_id | city   | street    | zipcode
----------+--------+-----------+--------
1         | 서울   | 강남대로  | 06000
1         | 부산   | 해운대로  | 48000
```

요소가 많으면 성능 이슈가 생기니까 변경이 잦으면 @OneToMany로 분리하는 게 낫다.

#### 언제 쓰고 언제 피할까

- 쓰기 좋은 경우
  - 변경이 거의 없는 단순 값 목록 (태그, 역할, 이메일 목록)
  - 별도 Entity로 만들기 애매한 작은 값 타입
- 피해야 하는 경우
  - 요소가 많고 자주 수정되는 컬렉션 → @OneToMany로 분리
  - 컬렉션 요소에 ID나 별도 쿼리가 필요한 경우 → 별도 Entity가 맞음
  - 요소별 LIKE/범위 쿼리 필요 → AttributeConverter와 같은 이슈

### 상속 관계 매핑

- @Inheritance — 상속 관계 매핑

JPA에서 Entity 상속 구조를 DB 테이블로 매핑하는 방법. 전략이 3가지 있음.

- 전략 1: SINGLE_TABLE (기본값)
   - 모든 자식 클래스를 하나의 테이블에 때려박고, DTYPE 컬럼으로 구분.
   - `DiscriminatorColumn`
장점: 조회 시 JOIN 없음, 성능 제일 좋음
단점: 다른 타입 컬럼은 전부 NULL 허용해야 함

- 전략 2: JOINED
   - 각 클래스마다 테이블 생성, 조회 시 JOIN.
장점: 정규화, NULL 없음, 명확한 구조
단점: 조회마다 JOIN → 성능 저하

- 전략 3: TABLE_PER_CLASS
  - 자식마다 완전히 독립된 테이블, 부모 컬럼 포함해서 전부 복사.
장점: 각 테이블 독립적, NULL 없음
단점: 부모 타입 조회 시 UNION ALL → 실무에서 거의 안 씀

## 4.5 애그리거트 영속성 전파

애그리거트 루트를 저장/삭제할 때 내부 구성요소도 함께 영속성 작업이 전파되는 것. `JPA의 CascadeType으로 제어`

```kotlin
@OneToMany(
    mappedBy = "order",
    cascade = [
        CascadeType.PERSIST,  // 저장 전파
        CascadeType.MERGE,    // 수정 전파
        CascadeType.REMOVE,   // 삭제 전파
    ]
)
// 위 세 개를 합친 게 ALL
cascade = [CascadeType.ALL]
```

@Embeddable 매핑 타입은 `cascade`속성 따로 지정하지 않아도 함께 저장되고 삭제된다.

## 4.7 도메인 구현과 DIP

우리가 만든 Entity는 @Entity, @Table, @Id등을 어노테이션을 활용하여 인프라에 의존하고 있다.

DDD에서 도메인 레이어는 인프라에 의존하면 안된다. 그런데 JPA, Redis 같은 인프라 기술 없이 어떻게 구현해야하는가?

![](https://velog.velcdn.com/images/cksgodl/post/aa3fc1a9-9f40-413f-9e51-11391e34eb63/image.png)

구현 기술에 대한 의존 없이 도메인을 순수하게 유지하려면 스프링 데이터 JPA의 레포지토리를 쓰는 것이 아니라, `OrderRepositoriy`를 구현한 `OrderRepositoriyImple`을 인프라 레이어에 두어야 한다.

이러한 구현은 과하다. 편한데로 쓰자.
